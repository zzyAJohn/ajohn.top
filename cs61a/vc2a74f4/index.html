通过编写样例测试来检查程序

def fbnc(x):
    """
    返回斐波那契数值
    >>> fbnc(4)
    2
    >>> fbnc(5)
    3
    """
    a, b = 0, 1
    n = 1
    while n < x:
        n += 1
        a, b = b, a + b
    return a
